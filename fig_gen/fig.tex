%   Usage: \node[fnode={<color>}, below=Xmm of <stage>] (<name>) {<content>};
\tikzset{
  fnode/.style n args={1}{
    rounded corners=4pt,
    draw=#1!45, fill=#1!7,
    font=\tiny\ttfamily, text=black!72,
    align=left,
    inner xsep=5pt, inner ysep=4pt,
    line width=0.65pt,
    text width=3.1cm
  },
  fnarr/.style={
    -{Latex[length=1.5mm,width=1.1mm]},
    line width=0.6pt, draw=black!28
  }
}

% ── API badge ─────────────────────────────────────────────────
\newcommand{\apibadge}[2]{%
  \tikz[baseline=(n.base)]\node[
    rounded corners=2pt,
    fill=#1!12, draw=#1!50, line width=0.6pt,
    font=\bfseries\scriptsize, text=#1!80!black,
    inner xsep=4pt, inner ysep=2pt
  ](n){#2};%
}

% ── Stage box (compact: title + description + badge, no files) ─
%   #1 (optional, default {}): extra tcolorbox options
%   #2 (mandatory)           : colour key
\newtcolorbox{stagebox}[2][]{%
  enhanced,
  width=3.5cm,
  equal height group=stage,
  colback=#2!5,
  colframe=#2!80,
  arc=6pt,
  boxrule=1.2pt,
  top=5pt, bottom=6pt, left=5pt, right=5pt,
  boxsep=0pt,
  fontupper=\scriptsize,
  valign=top,
  halign=flush left,
  #1,%
}
